import numpy as np
from phenom.source import sase_pulse

x = np.linspace(-500e-06, 500e-06, 512)
y = np.linspace(-500e-06, 500e-06, 512)
t = np.linspace(-100e-15, 100e-15, 512)

src  = SASE_Source(x = x,
                              y = y,
                              t = t,
                              photon_energy = 9200.,
                              pulse_energy = 1e-03,
                              pulse_duration = 25e-15,
                              bandwidth = 1e-03,
                              sigma = 100e-06,
                              div = 1e-03,
                              x0 = 0.,
                              y0 = 0.,
                              t0 = 2e-15,
                              theta_x = 0.,
                              theta_y = 0.
           )
